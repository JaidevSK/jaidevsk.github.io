ALGOLANG Documentation

Variable Declaration

						
let <variable_name>:<datatype> = <value>; # Store -- Global Dict

Reassign

						
<variable_name> = <new_value>; # Only for mut

Operations

						
Binary Arithmetic: +, -, /, *, //, %/, ^
Logical:
 Unary: NOT(boolean)
 Binary: AND(boolean1, boolean2), OR(boolean1, boolean2), XOR(boolean1, boolean2)
Comparison Operators: ==, !=, >, <, >=, <=

Comments

						
Add Comments
# Single Line

IO

						
print(<variable_name>);
print("STRING", <variable_name>);

Conditional

						
if (CONDITION) {DO SOMETHING}
elif (CONDITION) {DO SOMETHING}
elif (CONDITION) {DO SOMETHING}
else {DO SOMETHING}

Loops

						
while (CONDITION){
DO SOMETHING;
}
  • To break out of a loop, use the breakloop keyword.
  • To skip the rest of the code in a loop and continue to the next iteration, use the skipit keyword.

Functions (Alg)

						
let <function_name> = alg(<parameter>){
  <function_body>
};

Return

						
return <value>;

Function Call

						
let <variable_name> = <function_name>(<parameter>);

Global Dictionary

						
let <variable_name> = dict();

Project Euler Problem 3

						
let N = 600851475143;
#algo to find prime factors
let largest_prime = alg(N){
    let p = 2;
    let biggest = 1;
    while p^2 <= N{
        if N mod p == 0 {
            N = N div p;
            biggest = p;
            skipit;
        }
        p = p+1;
    }
    if N > 1 {
        biggest = N; # N is prime
    }
    return biggest;
};
print largest_prime(N);
